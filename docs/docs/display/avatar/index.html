export const App = component$(() => {
  return (
    <Avatar
        src="https://example.com/avatar.jpg"
        alt="User Avatar"
        size={4}
        sizeType="rem"
        shape="circle"
        fallback="UA"
        fallbackVariant="text"
        fallbackColor="success"
        width={100}
        height={100}
      />
  );
});
PropertyTypeDescription
classstringCustom CSS class for the component. (default: `undefined`)
styleCSSPropertiesInline styles for the component. (default: `undefined`)
srcstringURL of the image to be displayed in the avatar. (default: `undefined`)
altstringAlt text for the image. (default: `'avatar'`)
sizenumberSize of the avatar. (default: `2`)
sizeType'rem' | 'rm' | 'px' | stringUnit for the size (e.g., `rem`, `px`). (default: `'rem'`)
shape'circle' | 'square' | numberShape of the avatar (`'circle'`, `'square'`, or a percentage). (default: `'circle'`)
fallbackstringFallback text to display when the image fails to load. (default: `'Fb'`)
fallbackVariant'solid' | 'outlined' | 'text'Variant of the fallback style. (default: `'solid'`)
fallbackColor'success' | 'error' | 'warning' | 'accent' | 'primary'Color variant for the fallback style. (default: `'primary'`)
widthnumberWidth of the image (in pixels). (default: `900`)
heightnumberHeight of the image (in pixels). (default: `900`)
