Toggle Button Group Component for qwik

export const App = component$(() => {
    const handleToggle = $((value: boolean) => {
    console.log('Button toggled:', value);
  });

  const members: ToggleButtonMembers[] = [
    { id: 0, value: false, color: 'primary', onChange: handleToggle },
    { id: 1, value: true, color: 'success', onChange: handleToggle },
    { id: 2, value: false, color: 'error', onChange: handleToggle },
  ];
    
  return (
    <ToggleButtonGroup
      members={members}
      class="my-button-group"
      style={{ margin: "10px" }}
      variant="outlined"
      color="primary"
      rounded
      raised
      vertical
    >
      <h1 q:slot={"0"}>Toggle</h1>
      <h1 q:slot={"1"}>Toggle</h1>
      <h1 q:slot={"2"}>Toggle</h1>
    </ToggleButtonGroup>
  );
});

ToggleButtonMembers

PropertyTypeDescription
idnumberUnique identifier for the button.
disabledbooleanIf true, the button is disabled.
color"success", "error", "warning", "accent", "primary"The color of the button.
valuebooleanThe initial value of the button (true for active, false for inactive).
onChangeQRL(value: boolean) => voidCallback function triggered when the button's value changes.

Props

PropertyTypeDescription
membersToggleButtonMembers[]List of button members in the group.
classstringCustom class names to apply to the button group.
disabledbooleanIf true, all buttons in the group are disabled.
styleCSSPropertiesInline styles to apply to the button group.
color"success", "error", "warning", "accent", "primary"The color of the buttons in the group.
roundedbooleanIf true, the buttons in the group have rounded corners.
raisedbooleanIf true, the buttons in the group appear raised.
verticalbooleanIf true, the buttons are arranged vertically.
exclusivebooleanIf true, only one button can be active at a time.
