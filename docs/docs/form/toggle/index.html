Toggle Component for qwik

export const App = component$(() => {
  const state = useSignal(false);

  const handleChange = $((value) => {
    state.value = value;
  });

  return (
    <div>
      <Toggle
        class="my-toggle"
        style={{ margin: "10px" }}
        variant="slider"
        color="primary"
        size={3}
        sizeType="rem"
        onChange={handleChange}
        value={state.value}
      />
      <p>Toggle is {state.value ? 'On' : 'Off'}</p>
    </div>
  );
});
PropertyTypeDescription
classstringCustom class names to apply to the toggle container.
valuebooleanDefault value of the toggle switch.
disabledbooleanIf true, disables the toggle switch.
styleCSSPropertiesInline styles to apply to the toggle container.
variant"slider" | "radio" | "checkbox"Defines the toggle style variant. Defaults to "slider".
color"success" | "error" | "warning" | "accent" | "primary"Defines the color of the toggle switch. Defaults to "primary".
sizenumberSize of the toggle switch. Defaults to 2.
sizeType"rem" | "rm" | "px" | stringDefines the unit of size measurement. Defaults to "rem".
onChangeQRL<(value: boolean) => void>Event handler for when the toggle value changes.
