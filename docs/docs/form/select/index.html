export const App = component$(() => {
  const state = useSignal(undefined);

  const handleChange = $((value) => {
    state.value = value;
  });

  return (
    <div>
      <Select
        values={[
          { option: "Option 1", group: false },
          { option: "Option 2", group: false },
          { option: "Group 1", group: true },
          { option: "Option 3", group: false },
        ]}
        class="my-select"
        style={{ margin: "10px" }}
        variant="outlined"
        color="primary"
        rounded
        raised
        placeholder="Select an option"
        onChange={handleChange}
        maxHeight={150}
        direction="down"
      >
        <Button q:slot={"left"}>left</Button>
        <Button q:slot={"right"}>right</Button>
      </Select>
      <p>Selected Value: {state.value}</p>
    </div>
  );
});
PropertyTypeDescription
values{option: string; group: boolean}[]List of selectable options, where each option can be grouped.
valuestringDefault selected value of the select input.
classstringCustom class names to apply to the select container.
disabledbooleanIf true, disables the select input.
styleCSSPropertiesInline styles to apply to the select container.
color"success" | "error" | "warning" | "accent" | "primary"Defines the color of the select input. Defaults to "primary".
variant"solid" | "outlined" | "text"Defines the input style variant. Defaults to "outlined".
roundedbooleanIf true, applies rounded corners to the select input.
onChangeQRL<(value: string) => void>Event handler for when the select input value changes.
raisedbooleanIf true, applies a raised style to the select input.
placeholderstringPlaceholder text for the select input.
floatingPlaceholderbooleanIf true, displays a floating placeholder. Defaults to true.
maxHeightnumberMaximum height of the dropdown menu. Defaults to 100.
direction"down" | "up" | "left" | "right"Direction in which the dropdown menu will open. Defaults to "down".
